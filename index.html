async function verifyJob() { const jobNum = document.getElementById('jobNum').value.toUpperCase(); const empNum = document.getElementById('empNum').value; const btn = document.getElementById('verifyBtn'); const loader = document.getElementById('loader'); if (!jobNum) { alert("Please enter a job number"); return; } if (!empNum) { alert("Please enter your Employee Number"); return; } btn.disabled = true; loader.style.display = "block"; try { const res = await fetch(flowUrl, { method: "POST", headers: { "Content-Type": "application/json" }, // Action is now 'empverify' body: JSON.stringify({ action: "empverify", jobNum: jobNum, empNum: empNum }) }); const data = await res.json(); if (data.status === "found") { document.getElementById('regDisplay').innerText = data.registration; document.getElementById('descDisplay').innerText = data.description; document.getElementById('confirmSection').style.display = "block"; btn.style.display = "none"; } else { alert("Job not found or Employee ID invalid!"); } } catch (err) { alert("Connection error."); } finally { btn.disabled = false; loader.style.display = "none"; } }
